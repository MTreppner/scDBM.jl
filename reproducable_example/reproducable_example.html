Fit scDBM

Set up train and test set.

Random.seed!(101);
data, datatest = BMs.splitdata(pilot_data, 0.3);
datadict = BMs.DataDict("Training data" => data, "Test data" => datatest);

Set initial parameters. The learning rate has to be set relatively small, because the reconstruction can get very large for high expression values. Hence, the corresponding weights will get a very big learning signal.

epochs = 750;                                                        # Train for 5000 epochs
init_disp = (ones(size(data,2)) .* 1.0);                             # Set inverse dispersion parameter
lr = [0.000001*ones(3500);0.00005*ones(1000);0.00001*ones(10000)];   # Set learning rate
regularization = 220.0;                                              # Set regularization parameter (lambda)

Train scDBM.

monitor = BMs.Monitor(); monitor1 = BMs.Monitor(); monitor2 = BMs.Monitor();
Random.seed!(59);
dbm = BMs.fitdbm(data, epochs = 200, learningrate = 0.0001, batchsizepretraining = 32, 
      monitoringdatapretraining = datadict,
      pretraining = [
            # Negative-Binomial RBM
            BMs.TrainLayer(nhidden = 12,
            learningrates = lr, 
            epochs = epochs,
            rbmtype = BMs.NegativeBinomialBernoulliRBM, 
            inversedispersion = init_disp,
            fisherscoring = 1,
            lambda = regularization,
               monitoring = (rbm, epoch, datadict) -> begin if epoch % 20 == 0 BMs.monitorreconstructionerror!(monitor1, rbm, epoch, datadict) end end);
            # Bernoulli RBM
            BMs.TrainLayer(nhidden = 4, 
            learningrate = 0.0001,
            epochs = 500,
               monitoring = (rbm, epoch, datadict) -> begin if epoch % 20 == 0 BMs.monitorreconstructionerror!(monitor2, rbm, epoch, datadict) end end)
            ];
monitoring = (dbm, epoch) -> begin 
                              if epoch % 5 == 0 
                                 BMs.monitorlogproblowerbound!(monitor, dbm, epoch, datadict)
                              end
                           end);

pca_pilot_data = fit(PCA, pilot_data; maxoutdim=10);

# Map cells using Euclidean distance
maplab(x1,x2,y) = map(arg -> y[findmin(sum((x1 .- x2[arg:arg,:]).^2,dims=2))[2][1]],1:size(x2,1));
